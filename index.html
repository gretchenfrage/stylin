impl fmt::Display for Polynomial {
    #[allow(unused_must_use)]
    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
        if self.0.len() == 0 {
            return f.write_str("0");
        }

        fn digit_superscript(digit: char) -> char {
            match digit {
                '0' => '⁰',
                '1' => 'ⁱ',
                '2' => '²',
                '3' => '³',
                '4' => '⁴',
                '5' => '⁵',
                '6' => '⁶',
                '7' => '⁷',
                '8' => '⁸',
                '9' => '⁹',
                _ => panic!("unsupported digit for superscript: {}", digit)
            }
        }

        fn power_superscript(n: usize) -> String {
            n.to_string()
                .chars()
                .map(digit_superscript)
                .collect()
        }

        let mut buf = String::new();

        if self.0.len() > 1 {
            write!(&mut buf, "(");
        }

        let mut first_term_written = false;
        for (p, c) in self.0.iter().cloned().enumerate().rev() {
            if c == 0.0 {
                continue;
            }

            if first_term_written {
                if c > 0.0 {
                    write!(&mut buf, " + ");
                } else {
                    write!(&mut buf, " - ");
                }
            } else {
                if c < 0.0 {
                    write!(&mut buf, "-");
                }
                first_term_written = true;
            }

            if c.abs() == 1.0 {
                match p {
                    0 => { write!(&mut buf, "1"); },
                    1 => { write!(&mut buf, "x"); },
                    p => { write!(&mut buf, "x{}", power_superscript(p)); },
                };
            } else {
                write!(&mut buf, "{}", c.abs());
                match p {
                    0 => {},
                    1 => { write!(&mut buf, "x"); }
                    p => { write!(&mut buf, "x{}", power_superscript(p)); },
                };
            }
        }

        if self.0.len() > 1 {
            write!(&mut buf, ")");
        }

        f.write_str(&buf)
    }
}
